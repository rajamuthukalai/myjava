# Windows

# start zookeeper
C:\install\kafka\bin\windows>zookeeper-server-start.bat ../../config/zookeeper.properties

# start kafka server
C:\install\kafka\bin\windows>kafka-server-start.bat ../../config/server.properties

# create topic
C:\install\kafka\bin\windows>kafka-topics.bat --create --zookeeper localhost:2181 --replication-factor 1 --partitions 1 --topic customer

# start consumer
C:\install\kafka\bin\windows>kafka-console-consumer.bat --bootstrap-server localhost:9092 --topic customer --from-beginning
